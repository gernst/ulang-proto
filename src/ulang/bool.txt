definitions

not False == True;

True  or  False == True;
False or  True  == True;
False or  False == False;

True  ==> True  == True;
True  ==> False == False;
False ==> True  == True;
False ==> False == True;

(True  <=> True)  == True;
(True  <=> False) == False;
(False <=> True)  == False;
(False <=> False) == True;
